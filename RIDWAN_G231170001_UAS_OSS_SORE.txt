def Tampil_Tuple(tuple):
	for i in range(len(tuple)):
		print(i+1,'.',tuple[i])
#awalnya adalah kata tuple'e' kelebihan huruf e satu

def Tampil_List(list):
	for i in list:
		#diatas kata awalnya adalah listing diganti list saja
		print(i)
